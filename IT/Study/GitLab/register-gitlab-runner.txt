1. self-signed gitlab-server certs are trouble, do this:
  .get cert from gitlab-server
  # openssl s_client -connect alma9-01.local:443 </dev/null 2>/dev/null | sed -e '/-----BEGIN/,/-----END/!d' | sudo tee "$CERTIFICATE" > alma9-01.local.crt
  .create dir for the cert
  # mkdir /etc/gitlab-runner/certs
  .copy the cert to the dir, inlude hostname in the filename
  # cp gitlab.crt /etc/gitlab-runner/certs/alma9-01.local.crt
  .test:
  # echo|openssl s_client -CAfile /etc/gitlab-runner/certs/alma9-01.local.crt -connect alma9-01.local:443 -servername alma9-01.local:443
  : for more info see: https://docs.gitlab.com/runner/configuration/tls-self-signed/
